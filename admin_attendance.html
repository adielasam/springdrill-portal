window.loadAttendance = async () => { const tbody = document.getElementById('attendanceTableBody'); tbody.innerHTML = '
Fetching records...'; const selectedDate = document.getElementById('dateFilter').value; const selectedClass = document.getElementById('classFilter').value; try { // 1. Fetch raw data independently const [rawAtt, allStudents, allClasses] = await Promise.all([ supabase.from('attendance').select('*').order('created_at', { ascending: false }), supabase.from('students').select('*'), supabase.from('classes').select('*') ]); if (rawAtt.error) throw rawAtt.error; // 2. Map Classes & Students const classMap = {}; if (allClasses.data) allClasses.data.forEach(c => classMap[c.id] = c.name); // TRAP FIX: Map by ID, User_ID, AND Reg_Number! const studentMap = {}; if (allStudents.data) { allStudents.data.forEach(s => { s.className = classMap[s.class_id] || 'Unassigned'; studentMap[s.id] = s; if(s.user_id) studentMap[s.user_id] = s; if(s.reg_number) studentMap[s.reg_number] = s; }); } // 3. Combine them safely and bypass the Date Trap const filteredRecords = rawAtt.data.filter(record => { const sId = record.student_id || record.user_id || record.studentId || record.reg_number; const studentObj = studentMap[sId]; // DATE TRAP FIX: Force standardize the date format let dbDateString = ""; try { // Rely on the database's absolute timestamp to prevent MM/DD vs DD/MM confusion const rDate = new Date(record.created_at); const y = rDate.getFullYear(); const m = String(rDate.getMonth() + 1).padStart(2, '0'); const d = String(rDate.getDate()).padStart(2, '0'); dbDateString = `${y}-${m}-${d}`; } catch(e) { console.error(e); } const dateMatch = dbDateString === selectedDate; // Match Class (Fallbacks to ensure data is never hidden) const recClassId = record.class_id || (studentObj ? studentObj.class_id : 'unknown'); const classMatch = (selectedClass === 'all') || (recClassId == selectedClass); // Attach student object for rendering record.studentObj = studentObj; return dateMatch && classMatch; }); if (filteredRecords.length === 0) { tbody.innerHTML = '
No attendance records found. Ensure the correct date is selected or change filter to "All Classes".'; return; } tbody.innerHTML = filteredRecords.map(record => { const isPresent = String(record.status).toLowerCase().includes('present'); const badge = isPresent ? `Present` : `Absent`; // Fallback so it NEVER blanks out const rawID = record.student_id || record.reg_number || 'Unknown ID'; const studentName = record.studentObj ? `${record.studentObj.first_name} ${record.studentObj.surname || ''}` : `Unlinked Student (${rawID})`; const className = record.studentObj ? record.studentObj.className : 'Unknown Class'; const timeString = new Date(record.created_at).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); return ` ${studentName} ${className} ${timeString} ${badge} `; }).join(''); } catch (err) { console.error("Failed to load attendance:", err); tbody.innerHTML = `
Data Error: ${err.message}`; } };
